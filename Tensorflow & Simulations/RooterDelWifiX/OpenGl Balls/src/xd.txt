#include <iostream>
#include <cmath>
#include <math.h>
const float PI = M_PI;


void parametrizar(float x[], float y[]){
 int i = 0;
 float r = 1;
 for(float theta = 0; theta <= 2*PI; theta+= 0.01){
  x[i] = r*cos(theta);
  y[i] = r*sin(theta);
  i++;
 }
 x[i - 1] = r*cos(0);
 y[i - 1] = r*sin(0);
}


int main(){
  int l = floor(2*PI/0.01) + 1;
  float x[l];
  float y[l];
  parametrizar(x, y);
  for(int i = 0; i < l; i++){
    std::cout<<x[i]<<"  "<<y[i]<<std::endl;
  }
  return 0;
}
